Solution Exercice 2 – Module CommonJS

// math.js
function multiply(a, b) {
  return a * b;
}
module.exports = { multiply };

// app.js
const { multiply } = require('./math.js');
console.log(multiply(3, 4)); // 12

Solution Exercice 3 – Module ESM

// math.mjs
export function multiply(a, b) {
  return a * b;
}

// app.mjs
import { multiply } from './math.mjs';
console.log(multiply(3, 4)); // 12

Solution Exercice 5 – Script combiné (add & multiply)

// math.js
export function add(a, b) {
  return a + b;
}

export function multiply(a, b) {
  return a * b;
}

// app.mjs
import { add, multiply } from './math.js';

console.log("Addition : ", add(5, 7));       // 12
console.log("Multiplication : ", multiply(5, 7)); // 35
