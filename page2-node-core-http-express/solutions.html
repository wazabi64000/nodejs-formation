Solution Exercice 1

import http from 'http';

const server = http.createServer((req, res) => {
  res.writeHead(200, {'Content-Type':'text/plain'});
  res.end('Hello Node Core!');
});

server.listen(3000);

Solution Exercice 4

import http from 'http';
import url from 'url';

const server = http.createServer((req,res)=>{
  const query = url.parse(req.url,true).query;
  res.end('Paramètre: ' + query.name);
});

server.listen(3000);

Solution Exercice 5

import http from 'http';

const server = http.createServer((req,res)=>{
  if(req.method==='POST'){
    let body='';
    req.on('data',chunk=>body+=chunk);
    req.on('end',()=>{ res.end('Données reçues: ' + body); });
  } else {
    res.end('Envoyez une requête POST');
  }
});

server.listen(3000);
